# Заготовка для лабораторной работы 5 (датасет "Студенты")

import csv


with open("stud.csv", encoding='windows-1251') as file:
    csv_file = csv.reader(file, delimiter=";")
    header = next(csv_file, None)  # пропустить заголовки
    data_str = [row for row in csv_file]
# список data_str содержит списки значений признаков в каждой строке

data_size = len(data_str)  # число наблюдений
num_cols = len(data_str[0])  # число столбцов (общее число признаков)

# data_x - список списков из атрибутов, расположенных в каждой строке файла
data_x = [[] for _ in range(data_size)]
for i in range(data_size):
    data_x[i] = data_str[i].copy()
x_len = num_cols  # число входных признаков
x_names = header[:]  # имена входных признаков

print("Наблюдений:", data_size)
print("Входных признаков:", x_len)
print(x_names)
print()



print("Правда ли, что все студенты старше 2 курса старше 18 лет?")
val = True  # значение высказывания
violators_2course = []  # список нарушителей
for i in range(data_size):  # перечисляем строки таблицы, i - индекс строки
    # проверяем, нарушает ли i-й студент квантор общности
    # если нарушает, то присвоим значению высказывания val = False
    age_str = data_x[i][0]  # столбец с возрастом имеет индекс 0
    if age_str[0:5] == 'более':
        age_num = 23
    elif age_str[0:5] == 'менее':
        age_num = 16
    else:
        age_num = int(age_str[0:2])
    course_str = data_x[i][1]  # столбец с курсом имеет индекс 1
    # здесь мы обратились к элементу таблицы data_x в строке i в столбце 1
    if course_str[2:6] == 'курс':
        course_num = int(course_str[0])
        if course_num > 2:
            if age_num <= 18:
                val = False
                violators_2course.append(data_x[i])
print("Ответ:", val)
if violators_2course:
    print("Студенты старше 2 курса, нарушающие утверждение (возраст ≤ 18 лет):")
    for v in violators_2course:
        print(f"  {v}")
else:
    print("Нарушителей не найдено")
print()

print("Правда ли, что все студенты старше 1 курса старше 18 лет?")
val = True  # значение высказывания

# Задание 1
# Ответьте на поставленный вопрос, выведите строки, нарушающие квантор общности
# Для вывода i-й строки используйте оператор print(data_x[i])

violators = []  # список нарушителей
for i in range(data_size):
    age_str = data_x[i][0]
    if age_str[0:5] == 'более':
        age_num = 23
    elif age_str[0:5] == 'менее':
        age_num = 16
    else:
        age_num = int(age_str[0:2])

    course_str = data_x[i][1]
    # Проверяем, что студент старше 1 курса (не 1-й курс)
    if '1 курс' not in course_str:
        # Если студент старше 1 курса, но не старше 18 лет
        if age_num <= 18:
            val = False
            violators.append(data_x[i])

print("Ответ:", val)
if violators:
    print("Студенты старше 1 курса, нарушающие утверждение (возраст ≤ 18 лет):")
    for v in violators:
        print(f"  {v}")
else:
    print("Нарушителей не найдено")
print()

print("Правда ли, что найдутся студенты бакалавриата, которым менее 17 лет?")
val = False  # значение высказывания
found_students = []  # список найденных студентов
for i in range(data_size):
    # проверяем, удовлетворяет ли i-й студент условию квантора существования
    # если удовлетворяет, то присвоим значению высказывания val = True
    age_str = data_x[i][0]
    if age_str[0:5] == 'более':
        age_num = 23
    elif age_str[0:5] == 'менее':
        age_num = 16
    else:
        age_num = int(age_str[0:2])
    course_str = data_x[i][1]  # столбец с курсом имеет индекс 1
    if 'бакалавриат' in course_str:
        if age_num < 17:  # менее 17 лет
            val = True
            found_students.append(data_x[i])

print("Ответ:", val)
if found_students:
    print("Студенты бакалавриата, которым менее 17 лет:")
    for s in found_students:
        print(f"  {s}")
else:
    print("Таких студентов не найдено")
print()

print("Правда ли, что найдется студент, которому 21 год и который имеет стаж работы от 2 до 3 лет?")
val = False  # значение высказывания

# Задание 2
# Ответьте на поставленный вопрос, выведите строки, удовлетворяющие условию квантора существования
# Для вывода i-й строки используйте оператор print(data_x[i])

found_students_21 = []
for i in range(data_size):
    age_str = data_x[i][0]
    # Проверяем возраст 21 год
    if age_str == '21 год' or age_str == '21 лет':
        # Проверяем стаж работы
        experience_str = data_x[i][3]  # столбец с общим стажем работы имеет индекс 3
        if experience_str == 'от 2 до 3 лет':
            val = True
            found_students_21.append(data_x[i])

print("Ответ:", val)
if found_students_21:
    print("Студенты, которым 21 год и стаж работы от 2 до 3 лет:")
    for s in found_students_21:
        print(f"  {s}")
else:
    print("Таких студентов не найдено")
print()

# Дополнительный анализ: статистика по возрастам
print("Дополнительная статистика")
age_stats = {}
for i in range(data_size):
    age_str = data_x[i][0]
    age_stats[age_str] = age_stats.get(age_str, 0) + 1
print("Распределение по возрастам:")
for age, count in sorted(age_stats.items()):
    print(f"  {age}: {count}")

# Статистика по курсам
course_stats = {}
for i in range(data_size):
    course_str = data_x[i][1]
    # Извлекаем основной курс (без уточнений)
    if 'курс' in course_str:
        if 'бакалавриат' in course_str:
            main_course = 'бакалавриат'
        elif 'магистратура' in course_str:
            main_course = 'магистратура'
        elif 'специалитет' in course_str:
            main_course = 'специалитет'
        elif 'колледж' in course_str or 'коллелжа' in course_str:
            main_course = 'колледж'
        else:
            main_course = course_str[:10]
    else:
        main_course = 'другое'
    course_stats[main_course] = course_stats.get(main_course, 0) + 1
print("\nРаспределение по типам обучения:")
for course, count in sorted(course_stats.items()):
    print(f"  {course}: {count}")

# Статистика по стажу работы
experience_stats = {}
for i in range(data_size):
    exp_str = data_x[i][3]
    experience_stats[exp_str] = experience_stats.get(exp_str, 0) + 1
print("\nРаспределение по стажу работы:")
for exp, count in sorted(experience_stats.items()):
    print(f"  {exp}: {count}")
